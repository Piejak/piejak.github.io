                      def translateToMorse(in_string):
                          trans_dict = {'/': '-..-.', '=': '-...-', '3': '...--', 'Z': '--..', '1': '.----', 'X': '-..-', 'I': '..',
                                        'O': '---', 'R': '.-.', 'N': '-.', '9': '----.', '2': '..--', 'G': '--.', '@': '.--.-.', '0': '-----',
                                        'H': '....', '(': '-.--.-', 'U': '..-', "'": '.----.', 'Y': '-.--', 'K': '-.-', 'A': '.-',
                                        'L': '.-..', 'F': '..-.', '8': '---..', 'E': '.', 'M': '--', '5': '.....', 'T': '-', 'P': '.--.',
                                        '-': '-....-', '.': '.-.-.-', 'S': '...', '4': '....-', '7': '--...', ':': '---...', 'D': '-..',
                                        '?': '..--..', 'B': '-...', 'W': '.--', '"': '.-..-.', 'Q': '--.-', 'C': '-.-.', '6': '-....',
                                        ',': '--..--', 'J': '.---', 'V': '...-', ' ': '/', ')': '-.--.-'}
                          out_string = ""
                          for letter in in_string:
                              try:
                                  out_string += trans_dict[letter.upper()] + ' '
                              except KeyError:
                                  return letter + ' could not be translated into morse code. Try again'

                          return out_string
